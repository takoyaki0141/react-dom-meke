<head>
  <meta charset="UTF-8">
  <title>固定タブ＆モーダル</title>
  <style>
    /* タブのスタイル */
    #sideTab {
      position: fixed;
      top: 50%;
      right: 0;
      transform: translateY(-50%);
      background-color: #3498db;
      color: white;
      padding: 10px 15px;
      cursor: pointer;
      border-radius: 5px 0 0 5px;
      font-weight: bold;
    }

    /* モーダルの背景 */
    #modalBackground {
      display: none;
      position: fixed;
      top: 0; left: 0;
      width: 100%; height: 100%;
      background-color: rgba(0, 0, 0, 0.5);
      justify-content: center;
      align-items: center;
      z-index: 999;
    }

    /* モーダル本体 */
    #modalContent {
      background-color: white;
      padding: 20px;
      border-radius: 8px;
      min-width: 300px;
      text-align: center;
    }

    #closeBtn {
      margin-top: 15px;
      padding: 8px 16px;
      background-color: #e74c3c;
      color: white;
      border: none;
      border-radius: 4px;
      cursor: pointer;
    }
  </style>
</head>
<body>

<!-- 固定タブ -->
<div id="sideTab">モーダル開く</div>

<!-- モーダル -->
<div id="modalBackground">
  <div id="modalContent">
    <p>これはモーダルです！</p>
    <button id="closeBtn">閉じる</button>
  </div>
</div>

<script>
  const tab = document.getElementById("sideTab");
  const modalBg = document.getElementById("modalBackground");
  const closeBtn = document.getElementById("closeBtn");

  tab.addEventListener("click", () => {
    modalBg.style.display = "flex";
  });

  closeBtn.addEventListener("click", () => {
    modalBg.style.display = "none";
  });

  // モーダル背景クリックでも閉じる
  modalBg.addEventListener("click", (e) => {
    if (e.target === modalBg) {
      modalBg.style.display = "none";
    }
  });
</script>

</body>
